getInstanceType

该getInstanceType()方法返回该实例对象的类型。

[1,2,3].getInstanceType(); // "Array" {a:1,b:2}.getInstanceType(); // "Object" "abcdefg".getInstanceType(); // "String" var count = 12; count.getInstanceType(); // "Number"

assignPolyfill

该Object.assign()方法用于将所有可枚举的属性的值从一个或多个源对象复制到目标对象。它将返回目标对象。

Object.assign（target，... sources）

例子

var o1 = { a: 1, b: 1, c: 1 }; var o2 = { b: 2, c: 2 }; var o3 = { c: 3 }; var obj = Object.assign({}, o1, o2, o3); console.log(obj); // { a: 1, b: 2, c: 3 }

extend

该extend()方法用于将所有可枚举的属性的值从一个或多个源对象复制到目标对象。

var o1 = { a: 1, b: 1, c: 1 }; var o2 = { b: 2, c: 2 }; var o3 = { c: 3 }; var obj = {}.extend(o1, o2, o3); console.log(obj); // { a: 1, b: 2, c: 3 }

objectToParams

该objectToParams()方法用于将对象转为格式化的参数字符串。

var obj = { id: 100, page: 15, limit: 20 } var param = obj.objectToParams() console.log(param) // "id=100&page=15&limit=20"

countProperties

该countProperties()方法用于获取对象属性总数。

var obj = { a:1, b:2, c:3, d:4 } var count = obj.countProperties() console.log(count) // 4

each

该each()方法以 for.../for...in 循环遍历 Object 对象/Array 数组/类数组的可枚举属性，并以 callback 方式返回。

var obj = { a:1, b:2, c:3 } obj.each(function (value, key) { console.log(value, key) }) // a 1 // b 2 // c 3 var arr = ["a","b","c"] arr.each(function (index, value) { console.log(index, value) }) // 0 "a" // 1 "b" // 2 "c"

equals

该equals()方法用于循环比对两个目标对象的可枚举属性，以判断两个对象是否相等。

var obj1 = { a:1, b:2, c:3 } var obj2 = { c:3, b:2, a:1 } console.log(obj1.equals(obj2)) // true console.log(Object.equals(obj1, obj2)) // true

watch

该watch()方法用于监视对象属性的重新赋值。

var obj = { a: 5, b: 4} obj.watch("b", function (prop, oldValue, newValue) { console.log(prop, oldValue, newValue) }) obj.b = 7 // b 4 7

unwatch

该unwatch()方法用于解除对象的 watch 事件。

var obj = { a: 5, b: 4} obj.watch("b", function (prop, oldValue, newValue) { console.log(prop, oldValue, newValue) }) obj.unwatch("b") // 解除watch obj.b = 7

initial

该initial()方法用于初始化没有给定对象的默认值。

var obj = {} obj.initial('list', []).push([1,2,3]) console.log(obj) // {list:[1,2,3]}

setValueOfHref

该setValueOfHref()方法用于以解析字符 href 的方式给该对象赋值。

var obj = { item: { a : 5, b : 8, i : "b" } } obj.setValueOfHref('item.b', 3) console.log(obj.item.b) // 3 obj.setValueOfHref('item[item.i]', 7) console.log(obj.item.b) // 7

getValueByLink

该getValueByLink()方法用于以该对象为 context SCOPE 作为解析字符 link 的方式获取该对象字符表达式的计算值。

var obj = { item: { a : 5, b : 8, i : "b" }, limit: 100 } obj.getValueByRoute('item.b') // 8 obj.getValueByRoute('item[item.i]') // 8 obj.getValueByRoute('item[item.i] + item.a + limit') // 113
